//Цикл з постумовою
#include <iostream>
int main()
{
	int i = 1, n = 0;
	long int fact = 1;
	std::cout << "n = ";
	std::cin >> n;
	std::cout << std::endl;
	do{
		i = i + 1;
		fact = fact * i;
	}
	while (i < n);

	std::cout << "n! = " << fact << std::endl;
	std::cin.ignore(200,'\n').get();
	return 0;
}
