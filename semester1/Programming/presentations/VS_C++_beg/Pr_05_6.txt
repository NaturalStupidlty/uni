#include <iostream>
int main()
{
	int i,j;
	std::cout << "... 0%";
	for(i=5; i < 101; i+=5)
	{
	  for(j=0; j < 100000000; j++);//імітація процесу
	  std::cout << "\r... " << i << "%";
	};
	std::cout << std::endl;
	std::cin.ignore().get();
	/*std::cin.ignore(200,'\n').get();*/
	return 0;
}
